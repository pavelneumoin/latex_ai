% --- GraphZone: прямоугольник с клеткой для чертежа ---
\newcommand{\GraphZone}[2]{% #1=width in mm, #2=height in mm
  \begin{tikzpicture}
    % фон
    \fill[white] (0,0) rectangle (#1 mm, #2 mm);
    % клетка
    \foreach \x in {0,5,...,#1}
      \draw[gridline, line width=0.2pt] (\x mm, 0) -- (\x mm, #2 mm);
    \foreach \y in {0,5,...,#2}
      \draw[gridline, line width=0.2pt] (0, \y mm) -- (#1 mm, \y mm);
    % центральные оси
    \draw[gridaxis, line width=0.5pt, -{Stealth[length=2mm]}]
      (0, #2/2 mm) -- (#1 mm, #2/2 mm);
    \draw[gridaxis, line width=0.5pt, -{Stealth[length=2mm]}]
      (#1/2 mm, 0) -- (#1/2 mm, #2 mm);
    \node[anchor=south west, font=\sffamily\tiny, text=gridaxis]
      at (#1 mm - 3mm, #2/2 mm + 0.5mm) {$x$};
    \node[anchor=south west, font=\sffamily\tiny, text=gridaxis]
      at (#1/2 mm + 0.5mm, #2 mm - 3mm) {$y$};
    % рамка
    \draw[wcprimary, line width=0.6pt] (0,0) rectangle (#1 mm, #2 mm);
  \end{tikzpicture}}
