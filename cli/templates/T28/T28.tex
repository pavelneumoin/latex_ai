% Ячейка бинго: квадрат с угловыми диагоналями и AnswerField внизу.
\newcommand{\BingoCell}[3]{% #1 номер #2 задача #3 цвет угла
  \begin{tikzpicture}[baseline=(C.center)]
    \node[draw=wcprimary, line width=0.9pt, rounded corners=2pt,
          inner sep=2mm, minimum width=38mm, minimum height=38mm,
          align=center] (C) {};
    % Угловые диагональки
    \draw[#3, line width=1.3pt] (C.north west) -- ++(4mm,-4mm);
    \draw[#3, line width=1.3pt] (C.north east) -- ++(-4mm,-4mm);
    \draw[#3, line width=1.3pt] (C.south west) -- ++(4mm,4mm);
    \draw[#3, line width=1.3pt] (C.south east) -- ++(-4mm,4mm);
    % Номер
    \node[anchor=north west, font=\bfseries\footnotesize\color{wcprimary}]
      at ([xshift=1.5mm,yshift=-1.5mm]C.north west) {\##1};
    % Условие
    \node[font=\small, align=center, text width=32mm]
      at ([yshift=2mm]C.center) {#2};
    % Поле ответа
    \node[anchor=south, yshift=2mm] at (C.south)
      {\AnswerField{#1}};
  \end{tikzpicture}}
