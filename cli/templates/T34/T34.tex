% Большая флешкарта: верх (условие), линия-сгиб, низ (ответ крупно)
\newcommand{\Flashcard}[3]{% #1 номер #2 условие #3 высота условия в мм
  \par\vspace{2mm}\noindent\begin{tikzpicture}
    % Внешняя рамка
    \draw[wcprimary, line width=1.1pt, rounded corners=4pt]
      (0,0) rectangle (\linewidth, -#3mm);
    % Метка номера
    \node[anchor=north west, fill=wcprimary, text=white,
          font=\bfseries\sffamily, inner sep=2.5mm, rounded corners=1pt]
      at (2mm, -2mm) {КАРТА \##1};
    % Условие
    \node[anchor=north west, text width=\dimexpr\linewidth-8mm\relax,
          font=\large\sffamily, inner sep=4mm, yshift=-8mm]
      at (0,0) {#2};
    % Линия сгиба
    \draw[wcdivider, line width=0.9pt, dashed]
      (4mm, -\dimexpr#3mm-22mm\relax) -- (\linewidth-4mm, -\dimexpr#3mm-22mm\relax);
    \node[anchor=center, fill=wcprimaryBG, text=wcdivider,
          font=\scriptsize\itshape, inner sep=1mm]
      at (15mm, -\dimexpr#3mm-22mm\relax) {ОТВЕТ ↓};
    % Поле ответа (крупное, по центру нижней зоны)
    \node[anchor=center, font=\bfseries\large\sffamily\color{wcprimary}]
      at (\linewidth/2, -\dimexpr#3mm-11mm\relax)
      {Ответ: \AnswerField{#1}};
  \end{tikzpicture}\par}
